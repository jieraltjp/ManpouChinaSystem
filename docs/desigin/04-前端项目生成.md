```typescript
/**
 * API 统一响应结构。
 * 与后端 Result<T> 完全对应。
 */
export interface ApiResponse<T = unknown> {
  code: string
  message: string
  data: T | null
  traceId: string | null
  detail: string | null
}

/** 判断 API 响应是否成功 */
export function isSuccess<T>(res: ApiResponse<T>): res is ApiResponse<T> & { data: T } {
  return res.code === 'ok' && res.data !== null
}
```

## 4.4 创建 API 适配器

### 4.4.1 Axios 客户端

**`apps/web/apps/web/src/api/client.ts`**：

```typescript
/**
 * Axios 实例（自动注入 JWT + 错误处理）。
 */
import axios, { type AxiosInstance, type AxiosError } from 'axios'
import { useAuthStore } from '@/stores/auth'

const client: AxiosInstance = axios.create({
  baseURL: '/api/v1',
  timeout: 10_000,
  headers: {
    'Content-Type': 'application/json',
  },
})

// 请求拦截：注入 Access Token
client.interceptors.request.use(
  (config) => {
    const auth = useAuthStore()
    if (auth.token) {
      config.headers.Authorization = `Bearer ${auth.token}`
    }
    return config
  },
  (error) => Promise.reject(error),
)

// 响应拦截：统一错误处理
client.interceptors.response.use(
  (response) => response,
  async (error: AxiosError) => {
    const auth = useAuthStore()

    if (!error.response) {
      // 网络错误
      ElMessage.error('网络错误，请检查网络连接')
      return Promise.reject(error)
    }

    const { status, data } = error.response as { status: number; data: { message?: string } }
    const message = data?.message || error.message

    switch (status) {
      case 401:
        auth.logout()
        window.location.href = '/login'
        ElMessage.error('登录已过期，请重新登录')
        break
      case 403:
        ElMessage.error('没有权限访问此资源')
        break
      case 404:
        ElMessage.error('资源不存在')
        break
      case 500:
        ElMessage.error(`服务器错误: ${message}`)
        break
      default:
        ElMessage.error(message)
    }

    return Promise.reject(error)
  },
)

export default client
```

```typescript
/**
 * 认证状态管理。
 */
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { UserVO, LoginCmd } from '@/types/user'
import { login as loginApi, getCurrentUser } from '@/api/adapters/user'

export const useAuthStore = defineStore('auth', () => {
  // 状态
  const token = ref<string | null>(localStorage.getItem('token'))
  const user = ref<UserVO | null>(null)
  const loading = ref(false)

  // 计算属性
  const isAuthenticated = computed(() => !!token.value)
  const isAdmin = computed(() => user.value?.roles.includes('ADMIN'))
  const isManager = computed(() => user.value?.roles.includes('MANAGER'))

  // Actions
  async function login(credentials: LoginCmd): Promise<void> {
    loading.value = true
    try {
      const result = await loginApi(credentials)
      token.value = result.accessToken
      localStorage.setItem('token', result.accessToken)
      await fetchUser()
    } finally {
      loading.value = false
    }
  }

  async function fetchUser(): Promise<void> {
    if (!token.value) return
    try {
      user.value = await getCurrentUser()
    } catch {
      logout()
    }
  }

  function logout(): void {
    token.value = null
    user.value = null
    localStorage.removeItem('token')
  }

  // 初始化
  if (token.value) {
    fetchUser()
  }

  return {
    // 状态
    token,
    user,
    loading,
    // 计算属性
    isAuthenticated,
    isAdmin,
    isManager,
    // Actions
    login,
    logout,
    fetchUser,
  }
})
```

### 4.5.2 用户状态

**`apps/web/apps/web/src/stores/user.ts`**：

```typescript
/**
 * 用户状态管理。
 */
import { defineStore } from 'pinia'
import { ref } from 'vue'
import type { UserVO, UserQuery } from '@/types/user'
import {
  queryUsers,
  createUser,
  updateUser,
  deleteUser,
} from '@/api/adapters/user'

export const useUserStore = defineStore('user', () => {
  const users = ref<UserVO[]>([])
  const total = ref(0)
  const loading = ref(false)

  async function fetchUsers(query: UserQuery): Promise<void> {
    loading.value = true
    try {
      const result = await queryUsers(query)
      users.value = result.content
      total.value = result.totalElements
    } finally {
      loading.value = false
    }
  }

  async function create(cmd: Omit<UserVO, 'id'>): Promise<UserVO> {
    const user = await createUser(cmd as any)
    users.value.unshift(user)
    return user
  }

  async function update(id: number, cmd: Partial<UserVO>): Promise<UserVO> {
    const user = await updateUser(id, cmd as any)
    const index = users.value.findIndex((u) => u.id === id)
    if (index !== -1) {
      users.value[index] = user
    }
    return user
  }

  async function remove(id: number): Promise<void> {
    await deleteUser(id)
    users.value = users.value.filter((u) => u.id !== id)
  }

  return {
    users,
    total,
    loading,
    fetchUsers,
    create,
    update,
    remove,
  }
})
```

```typescript
const router = createRouter({

  routes,

// 路由守卫
router.beforeEach((to, _from, next) => {
  const auth = useAuthStore()

  // 公开路由
  if (to.meta.public) {
    if (auth.isAuthenticated && to.path === '/login') {
      next('/dashboard')
    } else {
      next()
    }
    return
  }

  // 需要认证
  if (!auth.isAuthenticated) {
    next('/login')
    return
  }

  // 角色权限检查
  const requiredRoles = to.meta.roles as string[] | undefined
  if (requiredRoles?.length) {
    const hasRole = requiredRoles.some((role) => auth.user?.roles.includes(role))
    if (!hasRole) {
      next('/dashboard')
      return
    }
  }

  next()
})

export default router
```
