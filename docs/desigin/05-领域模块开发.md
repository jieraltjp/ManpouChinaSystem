```java
package com.manpou.user.domain.model;

import com.manpou.user.domain.model.base.BaseEntity;
import jakarta.persistence.*;
import lombok.Getter;
import lombok.NoArgsConstructor;

import java.time.LocalDateTime;

/**
 * 用户聚合根。
 *
 * 唯一标识：id
 *
 * 不变式：
 * - username 非空，长度 3-32，唯一
 * - email 格式正确
 * - phone 格式正确（11位手机号）
 * - password BCrypt 加密存储
 */
@Entity
@Table(name = "user", indexes = {
    @Index(name = "idx_username", columnList = "username"),
    @Index(name = "idx_email", columnList = "email"),
    @Index(name = "idx_status", columnList = "status")
})
@Getter
@NoArgsConstructor
public class User extends BaseEntity {

    @Column(name = "username", nullable = false, unique = true, length = 32)
    private String username;

    @Column(name = "password", nullable = false, length = 128)
    private String password;

    @Column(name = "email", nullable = false, length = 64)
    private String email;

    @Column(name = "phone", length = 20)
    private String phone;

    @Enumerated(EnumType.ORDINAL)
    @Column(name = "status", nullable = false)
    private UserStatus status = UserStatus.ACTIVE;

    // ========== 构造函数 ==========

    public User(String username, String password, String email, String phone) {
        this.username = username;
        this.password = password;
        this.email = email;
        this.phone = phone;
        this.status = UserStatus.ACTIVE;
    }

    // ========== 领域方法 ==========

    /** 禁用账户 */
    public void suspend() {
        if (this.status == UserStatus.DELETED) {
            throw new IllegalStateException("已删除的用户不能禁用");
        }
        this.status = UserStatus.SUSPENDED;
    }

    /** 激活账户 */
    public void activate() {
        if (this.status == UserStatus.DELETED) {
            throw new IllegalStateException("已删除的用户不能激活");
        }
        this.status = UserStatus.ACTIVE;
    }

    /** 逻辑删除 */
    @Override
    public void markDeleted() {
        this.status = UserStatus.DELETED;
        super.markDeleted();
    }

    /** 验证密码 */
    public boolean verifyPassword(String rawPassword, String encodedPassword) {
        // 使用 BCrypt 验证
        return org.springframework.security.crypto.bcrypt.BCryptPasswordEncoder.class
            .cast(null)
            .matches(rawPassword, encodedPassword);
    }

    /** 是否活跃 */
    public boolean isActive() {
        return this.status == UserStatus.ACTIVE;
    }
}
```

### 5.1.4 用户用例服务

**`apps/user-service/src/main/java/com/manpou/user/application/usecase/UserUseCase.java`**：

```java
package com.manpou.user.application.usecase;

import com.manpou.user.application.dto.UserCreateCmd;
import com.manpou.user.application.dto.UserUpdateCmd;
import com.manpou.user.application.dto.UserQuery;
import com.manpou.user.application.assembler.UserAssembler;
import com.manpou.user.domain.model.User;
import com.manpou.user.domain.model.UserStatus;
import com.manpou.user.domain.repository.UserRepository;
import com.manpou.common.exception.BusinessException;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.data.jpa.domain.Specification;
import org.springframework.security.crypto.bcrypt.BCryptPasswordEncoder;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import jakarta.persistence.criteria.Predicate;
import java.util.ArrayList;
import java.util.List;

/**
 * 用户用例服务。
 * 编排用户相关业务操作，保持领域模型纯净。
 */
@Slf4j
@Service
@RequiredArgsConstructor
public class UserUseCase {

    private final UserRepository userRepository;
    private final UserAssembler userAssembler;
    private final BCryptPasswordEncoder passwordEncoder;

    // ========== 命令方法 (写操作) ==========

    /**
     * 创建用户。
     */
    @Transactional
    public User createUser(UserCreateCmd cmd) {
        // 1. 唯一性检查
        if (userRepository.existsByUsername(cmd.getUsername())) {
            throw new BusinessException("user.username-exists", "用户名已存在");
        }
        if (userRepository.existsByEmail(cmd.getEmail())) {
            throw new BusinessException("user.email-exists", "邮箱已被注册");
        }

        // 2. 创建实体
        User user = userAssembler.toEntity(cmd);
        user.setPassword(passwordEncoder.encode(cmd.getPassword()));

        // 3. 保存
        User saved = userRepository.save(user);
        log.info("[User] created: id={}, username={}", saved.getId(), saved.getUsername());

        return saved;
    }

    /**
     * 更新用户信息。
     */
    @Transactional
    public User updateUser(Long id, UserUpdateCmd cmd) {
        User user = getUserById(id);

        // 更新字段
        if (cmd.getEmail() != null && !cmd.getEmail().equals(user.getEmail())) {
            if (userRepository.existsByEmail(cmd.getEmail())) {
                throw new BusinessException("user.email-exists", "邮箱已被注册");
            }
            user.setEmail(cmd.getEmail());
        }

        if (cmd.getPhone() != null) {
            user.setPhone(cmd.getPhone());
        }

        User saved = userRepository.save(user);
        log.info("[User] updated: id={}", id);

        return saved;
    }

    /**
     * 修改密码。
     */
    @Transactional
    public void changePassword(Long id, String oldPassword, String newPassword) {
        User user = getUserById(id);

        // 验证旧密码
        if (!passwordEncoder.matches(oldPassword, user.getPassword())) {
            throw new BusinessException("user.password-mismatch", "原密码错误");
        }

        // 设置新密码
        user.setPassword(passwordEncoder.encode(newPassword));
        userRepository.save(user);

        log.info("[User] password changed: id={}", id);
    }

    /**
     * 禁用用户。
     */
    @Transactional
    public void suspendUser(Long id) {
        User user = getUserById(id);
        user.suspend();
        userRepository.save(user);
        log.info("[User] suspended: id={}", id);
    }

    /**
     * 激活用户。
     */
    @Transactional
    public void activateUser(Long id) {
        User user = getUserById(id);
        user.activate();
        userRepository.save(user);
        log.info("[User] activated: id={}", id);
    }

    /**
     * 删除用户（逻辑删除）。
     */
    @Transactional
    public void deleteUser(Long id) {
        User user = getUserById(id);
        user.markDeleted();
        userRepository.save(user);
        log.info("[User] deleted: id={}", id);
    }

    // ========== 查询方法 (读操作) ==========

    /**
     * 根据 ID 获取用户。
     */
    @Transactional(readOnly = true)
    public User getUserById(Long id) {
        return userRepository.findById(id)
            .filter(u -> u.getStatus() != UserStatus.DELETED)
            .orElseThrow(() -> new BusinessException("user.not-found", "用户不存在"));
    }

    /**
     * 根据用户名获取用户。
     */
    @Transactional(readOnly = true)
    public User getUserByUsername(String username) {
        return userRepository.findByUsername(username)
            .filter(u -> u.getStatus() != UserStatus.DELETED)
            .orElseThrow(() -> new BusinessException("user.not-found", "用户不存在"));
    }

    /**
     * 分页查询用户。
     */
    @Transactional(readOnly = true)
    public Page<User> queryUsers(UserQuery query, Pageable pageable) {
        Specification<User> spec = (root, cq, cb) -> {
            List<Predicate> predicates = new ArrayList<>();

            // 用户名模糊查询
            if (query.getUsername() != null && !query.getUsername().isBlank()) {
                predicates.add(cb.like(root.get("username"), "%" + query.getUsername() + "%"));
            }

            // 邮箱精确查询
            if (query.getEmail() != null && !query.getEmail().isBlank()) {
                predicates.add(cb.equal(root.get("email"), query.getEmail()));
            }

            // 状态过滤
            if (query.getStatus() != null) {
                predicates.add(cb.equal(root.get("status"), query.getStatus()));
            } else {
                // 默认排除已删除
                predicates.add(cb.notEqual(root.get("status"), UserStatus.DELETED));
            }

            return cb.and(predicates.toArray(new Predicate[0]));
        };

        return userRepository.findAll(spec, pageable);
    }
}
```

```java
package com.manpou.user.interfaces.controller;

import com.manpou.user.application.dto.UserUpdateCmd;

import java.util.List;

/**
 * 用户管理 REST 接口。

@RestController
@RequestMapping("/users")
@RequiredArgsConstructor
public class UserController {

    private final UserUseCase userUseCase;

    // ========== 命令接口 ==========

    /**
     * 创建用户。
     */
    @PostMapping
    public Result<User> createUser(@RequestBody @Valid UserCreateCmd cmd) {

     * 更新用户信息。

    @PutMapping("/{id}")

        // 验证：只能修改自己的密码，或者管理员可以修改他人密码
        if (!id.toString().equals(userId)) {
            // TODO: 添加管理员权限检查
        }
        userUseCase.changePassword(id, cmd.getOldPassword(), cmd.getNewPassword());
        return Result.success();
    }

    /**
     * 禁用用户。
     */
    @PostMapping("/{id}/suspend")
    public Result<Void> suspendUser(@PathVariable Long id) {
        userUseCase.suspendUser(id);
        return Result.success();
    }

    /**
     * 激活用户。
     */
    @PostMapping("/{id}/activate")
    public Result<Void> activateUser(@PathVariable Long id) {
        userUseCase.activateUser(id);
        return Result.success();
    }

    /**
     * 删除用户。
     */
    @DeleteMapping("/{id}")
    public Result<Void> deleteUser(@PathVariable Long id) {
        userUseCase.deleteUser(id);
        return Result.success();
    }

    // ========== 查询接口 ==========

    /**
     * 获取当前登录用户。
     */
    @GetMapping("/me")
    public Result<User> getCurrentUser(@RequestHeader("X-User-Id") String userId) {
        User user = userUseCase.getUserById(Long.parseLong(userId));
        return Result.success(user);
    }

    /**
     * 根据 ID 获取用户。
     */
    @GetMapping("/{id}")
    public Result<User> getUserById(@PathVariable Long id) {
        User user = userUseCase.getUserById(id);
        return Result.success(user);
    }

    /**
     * 分页查询用户。
     */
    @GetMapping
    public Result<Page<User>> queryUsers(
            UserQuery query,
            @PageableDefault(size = 20) Pageable pageable) {
        Page<User> page = userUseCase.queryUsers(query, pageable);
        return Result.success(page);
    }

    // ========== 内部类 ==========

    public static class PasswordChangeCmd {
        private String oldPassword;
        private String newPassword;

        public String getOldPassword() { return oldPassword; }
        public void setOldPassword(String oldPassword) { this.oldPassword = oldPassword; }
        public String getNewPassword() { return newPassword; }
        public void setNewPassword(String newPassword) { this.newPassword = newPassword; }
    }
```

```java
package com.manpou.product.domain.model;

import com.manpou.product.domain.model.base.BaseEntity;
import jakarta.persistence.*;
import lombok.Getter;
import lombok.NoArgsConstructor;

import java.math.BigDecimal;
import java.util.ArrayList;
import java.util.List;

/**
 * 商品聚合根。
 *
 * 唯一标识：id
 *
 * 业务规则：
 * - 上架商品库存必须 >= 0
 * - 价格必须 > 0
 * - 下架后不能直接销售
 */
@Entity
@Table(name = "product", indexes = {
    @Index(name = "idx_name", columnList = "name"),
    @Index(name = "idx_category", columnList = "category_id"),
    @Index(name = "idx_status", columnList = "status")
})
@Getter
@NoArgsConstructor
public class Product extends BaseEntity {

    @Column(name = "name", nullable = false, length = 255)
    private String name;

    @Column(name = "description", columnDefinition = "TEXT")
    private String description;

    @Column(name = "price", nullable = false, precision = 19, scale = 2)
    private BigDecimal price;

    @Column(name = "cost", precision = 19, scale = 2)
    private BigDecimal cost;

    @Column(name = "stock", nullable = false)
    private Integer stock = 0;

    @Column(name = "category_id")
    private Long categoryId;

    @Column(name = "images", columnDefinition = "JSON")
    private String images; // JSON array

    @Enumerated(EnumType.ORDINAL)
    @Column(name = "status", nullable = false)
    private ProductStatus status = ProductStatus.DRAFT;

    // ========== 构造函数 ==========

    public Product(String name, BigDecimal price, Integer stock) {
        this.name = name;
        this.price = price;
        this.stock = stock;
        this.status = ProductStatus.DRAFT;
    }

    // ========== 领域方法 ==========

    /**
     * 上架商品。
     */
    public void activate() {
        if (this.stock < 0) {
            throw new IllegalStateException("库存不能为负数");
        }
        this.status = ProductStatus.ACTIVE;
    }

    /**
     * 下架商品。
     */
    public void offShelf() {
        this.status = ProductStatus.OFF_SHELF;
    }

    /**
     * 扣减库存。
     */
    public void decreaseStock(Integer quantity) {
        if (quantity <= 0) {
            throw new IllegalArgumentException("扣减数量必须大于 0");
        }
        if (this.stock < quantity) {
            throw new IllegalStateException("库存不足，当前库存: " + this.stock);
        }
        this.stock -= quantity;
    }

    /**
     * 增加库存。
     */
    public void increaseStock(Integer quantity) {
        if (quantity <= 0) {
            throw new IllegalArgumentException("增加数量必须大于 0");
        }
        this.stock += quantity;
    }

    /**
     * 是否可售。
     */
    public boolean isOnSale() {
        return this.status == ProductStatus.ACTIVE && this.stock > 0;
    }

    /**
     * 逻辑删除。
     */
    @Override
    public void markDeleted() {
        this.status = ProductStatus.DELETED;
        super.markDeleted();
    }
}
```

### 5.2.2 库存服务（领域服务）

**`apps/product-service/src/main/java/com/manpou/product/domain/service/InventoryService.java`**：

```java
package com.manpou.product.domain.service;

import com.manpou.product.domain.model.Product;
import com.manpou.product.domain.model.ProductStatus;
import com.manpou.product.domain.repository.ProductRepository;
import com.manpou.common.exception.BusinessException;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.util.List;

/**
 * 库存领域服务。
 * 处理跨商品的库存操作，封装业务规则。
 */
@Slf4j
@Service
@RequiredArgsConstructor
public class InventoryService {

    private final ProductRepository productRepository;

    /**
     * 批量扣减库存。
     *
     * @param items [(productId, quantity), ...]
     */
    @Transactional
    public void decreaseStockBatch(List<StockItem> items) {
        for (StockItem item : items) {
            Product product = productRepository.findById(item.productId())
                .orElseThrow(() -> new BusinessException(
                    "product.not-found",
                    "商品不存在: " + item.productId()));

            // 校验状态
            if (product.getStatus() != ProductStatus.ACTIVE) {
                throw new BusinessException(
                    "product.not-active",
                    "商品未上架: " + product.getName());
            }

            // 扣减库存
            product.decreaseStock(item.quantity());
            productRepository.save(product);

            log.info("[Inventory] decreased: productId={}, quantity={}, remaining={}",
                item.productId(), item.quantity(), product.getStock());
        }
    }

    /**
     * 回退库存（退款场景）。
     */
    @Transactional
    public void increaseStockBatch(List<StockItem> items) {
        for (StockItem item : items) {
            Product product = productRepository.findById(item.productId())
                .orElseThrow(() -> new BusinessException(
                    "product.not-found",
                    "商品不存在: " + item.productId()));

            product.increaseStock(item.quantity());
            productRepository.save(product);

            log.info("[Inventory] increased: productId={}, quantity={}, remaining={}",
                item.productId(), item.quantity(), product.getStock());
        }
    }

    /**
     * 库存预警检查。
     */
    @Transactional(readOnly = true)
    public List<Product> getLowStockProducts(int threshold) {
        return productRepository.findLowStockProducts(
            ProductStatus.ACTIVE, threshold);
    }

    // ========== 内部类 ==========

    public record StockItem(Long productId, Integer quantity) {}
}
```

**`apps/order-service/src/main/java/com/manpou/order/domain/model/Order.java`**：

```java
package com.manpou.order.domain.model;

import com.manpou.order.domain.model.base.BaseEntity;
import jakarta.persistence.*;
import lombok.Getter;
import lombok.NoArgsConstructor;

import java.math.BigDecimal;
import java.time.LocalDateTime;
import java.util.ArrayList;
import java.util.List;

/**
 * 订单聚合根。
 *
 * 唯一标识：id
 * 聚合边界：Order + OrderItem + Address
 *
 * 状态机：
 * PENDING -> PAID -> SHIPPED -> COMPLETED
 *         -> CANCELLED (PENDING 阶段可取消)
 *         -> REFUNDING -> REFUNDED (PAID 阶段可退款)
 */
@Entity
@Table(name = "`order`", indexes = {
    @Index(name = "idx_order_no", columnList = "order_no"),
    @Index(name = "idx_user_id", columnList = "user_id"),
    @Index(name = "idx_status", columnList = "status"),
    @Index(name = "idx_created_at", columnList = "create_time")
})
@Getter
@NoArgsConstructor
public class Order extends BaseEntity {

    @Column(name = "order_no", nullable = false, unique = true, length = 32)
    private String orderNo;

    @Column(name = "user_id", nullable = false)
    private Long userId;

    @OneToMany(cascade = CascadeType.ALL, orphanRemoval = true, fetch = FetchType.LAZY)
    @JoinColumn(name = "order_id", nullable = false)
    private List<OrderItem> items = new ArrayList<>();

    @Column(name = "total_amount", nullable = false, precision = 19, scale = 2)
    private BigDecimal totalAmount = BigDecimal.ZERO;

    @Column(name = "discount_amount", precision = 19, scale = 2)
    private BigDecimal discountAmount = BigDecimal.ZERO;

    @Column(name = "pay_amount", nullable = false, precision = 19, scale = 2)
    private BigDecimal payAmount = BigDecimal.ZERO;

    @Enumerated(EnumType.ORDINAL)
    @Column(name = "status", nullable = false)
    private OrderStatus status = OrderStatus.PENDING;

    @Column(name = "shipping_address", columnDefinition = "TEXT")
    private String shippingAddress; // JSON

    @Column(name = "remark", length = 500)
    private String remark;

    @Column(name = "paid_at")
    private LocalDateTime paidAt;

    @Column(name = "shipped_at")
    private LocalDateTime shippedAt;

    @Column(name = "completed_at")
    private LocalDateTime completedAt;

    @Column(name = "cancelled_at")
    private LocalDateTime cancelledAt;

    @Column(name = "cancel_reason", length = 255)
    private String cancelReason;

    // ========== 构造函数 ==========

    public Order(String orderNo, Long userId) {
        this.orderNo = orderNo;
        this.userId = userId;
        this.status = OrderStatus.PENDING;
    }

    // ========== 聚合根操作 ==========

    /**
     * 添加订单项。
     */
    public void addItem(OrderItem item) {
        items.add(item);
        recalculateAmount();
    }

    /**
     * 移除订单项。
     */
    public void removeItem(Long productId) {
        items.removeIf(item -> item.getProductId().equals(productId));
        recalculateAmount();
    }

    /**
     * 重新计算金额。
     */
    private void recalculateAmount() {
        this.totalAmount = items.stream()
            .map(OrderItem::getSubtotal)
            .reduce(BigDecimal.ZERO, BigDecimal::add);
        this.payAmount = this.totalAmount.subtract(
            this.discountAmount != null ? this.discountAmount : BigDecimal.ZERO);
    }

    /**
     * 支付成功。
     */
    public void pay() {
        if (this.status != OrderStatus.PENDING) {
            throw new IllegalStateException("只有待支付的订单可以支付");
        }
        this.status = OrderStatus.PAID;
        this.paidAt = LocalDateTime.now();
    }

    /**
     * 发货。
     */
    public void ship() {
        if (this.status != OrderStatus.PAID) {
            throw new IllegalStateException("只有已支付的订单可以发货");
        }
        this.status = OrderStatus.SHIPPED;
        this.shippedAt = LocalDateTime.now();
    }

    /**
     * 确认收货/完成。
     */
    public void complete() {
        if (this.status != OrderStatus.SHIPPED) {
            throw new IllegalStateException("只有已发货的订单可以确认收货");
        }
        this.status = OrderStatus.COMPLETED;
        this.completedAt = LocalDateTime.now();
    }

    /**
     * 取消订单。
     */
    public void cancel(String reason) {
        if (this.status != OrderStatus.PENDING) {
            throw new IllegalStateException("只有待支付的订单可以取消");
        }
        this.status = OrderStatus.CANCELLED;
        this.cancelledAt = LocalDateTime.now();
        this.cancelReason = reason;
    }

    // ========== 业务规则 ==========

    /** 是否可以取消 */
    public boolean canCancel() {
        return this.status == OrderStatus.PENDING;
    }

    /** 是否可以退款 */
    public boolean canRefund() {
        return this.status == OrderStatus.PAID || this.status == OrderStatus.SHIPPED;
    }

    /** 是否已支付 */
    public boolean isPaid() {
        return this.status != OrderStatus.PENDING && this.status != OrderStatus.CANCELLED;
    }
}
```

```java
package com.manpou.order.domain.model;

import lombok.Getter;

/**

 *
 * 设计为值对象的原因是：
 * 1. 无独立标识，随订单生命周期
 * 2. 不可变（除数量调整外）
 * 3. 包含商品快照（商品修改不影响历史订单）

@Entity
@Table(name = "order_item")
@Getter
@NoArgsConstructor
public class OrderItem {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Column(name = "order_id", nullable = false)
    private Long orderId;

    @Column(name = "product_id", nullable = false)
    private Long productId;

    @Column(name = "product_name", nullable = false, length = 255)
    private String productName;

    @Column(name = "sku_id")
    private Long skuId;

    @Column(name = "sku_snapshot", columnDefinition = "TEXT")
    private String skuSnapshot; // SKU 规格快照 JSON

    @Column(name = "quantity", nullable = false)
    private Integer quantity;

    @Column(name = "unit_price", nullable = false, precision = 19, scale = 2)
    private BigDecimal unitPrice;

    @Column(name = "discount", precision = 19, scale = 2)
    private BigDecimal discount = BigDecimal.ZERO;

    @Column(name = "subtotal", nullable = false, precision = 19, scale = 2)
    private BigDecimal subtotal;

    // ========== 构造函数 ==========

    public OrderItem(Long productId, String productName, Long skuId,
                     Integer quantity, BigDecimal unitPrice) {
        this.productId = productId;
        this.productName = productName;
        this.skuId = skuId;
        this.quantity = quantity;
        this.unitPrice = unitPrice;
        this.discount = BigDecimal.ZERO;

    }

    // ========== 方法 ==========

     * 获取折后小计。

    public BigDecimal getDiscountedSubtotal() {
```

### 5.3.3 订单创建用例

**`apps/order-service/src/main/java/com/manpou/order/application/usecase/OrderUseCase.java`**：

```java
package com.manpou.order.application.usecase;

import com.manpou.order.application.dto.OrderCreateCmd;
import com.manpou.order.application.dto.OrderQuery;
import com.manpou.order.application.event.OrderCreatedEvent;
import com.manpou.order.domain.model.Order;
import com.manpou.order.domain.model.OrderItem;
import com.manpou.order.domain.model.OrderStatus;
import com.manpou.order.domain.model.StockItem;
import com.manpou.order.domain.repository.OrderRepository;
import com.manpou.order.domain.service.InventoryService;
import com.manpou.product.domain.model.Product;
import com.manpou.product.domain.repository.ProductRepository;
import com.manpou.common.exception.BusinessException;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.context.ApplicationEventPublisher;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.math.BigDecimal;
import java.time.LocalDateTime;
import java.time.format.DateTimeFormatter;
import java.util.List;
import java.util.concurrent.atomic.AtomicLong;

/**
 * 订单用例服务。
 * 编排订单创建、支付、退款等核心业务流程。
 */
@Slf4j
@Service
@RequiredArgsConstructor
public class OrderUseCase {

    private final OrderRepository orderRepository;
    private final ProductRepository productRepository;
    private final InventoryService inventoryService;
    private final ApplicationEventPublisher eventPublisher;

    private static final AtomicLong orderSequence = new AtomicLong(1);
    private static final DateTimeFormatter ORDER_NO_FORMAT =
        DateTimeFormatter.ofPattern("yyyyMMddHHmmss");

    // ========== 创建订单 ==========

    /**
     * 创建订单。
     *
     * 核心流程：
     * 1. 校验商品信息
     * 2. 锁定库存（预扣减）
     * 3. 创建订单
     * 4. 发布领域事件
     */
    @Transactional
    public Order createOrder(Long userId, OrderCreateCmd cmd) {
        // 1. 校验并构建订单项
        List<OrderItem> items = buildOrderItems(cmd.getItems());
        List<StockItem> stockItems = items.stream()
            .map(item -> new StockItem(item.getProductId(), item.getQuantity()))
            .toList();

        // 2. 预锁定库存
        try {
            inventoryService.decreaseStockBatch(stockItems);
        } catch (BusinessException e) {
            log.warn("[Order] stock lock failed: userId={}", userId);
            throw e;
        }

        // 3. 创建订单
        String orderNo = generateOrderNo();
        Order order = new Order(orderNo, userId);
        order.setShippingAddress(cmd.getShippingAddress());
        order.setRemark(cmd.getRemark());

        items.forEach(order::addItem);
        order.setDiscountAmount(cmd.getDiscountAmount() != null
            ? cmd.getDiscountAmount() : BigDecimal.ZERO);

        Order saved = orderRepository.save(order);
        log.info("[Order] created: orderNo={}, userId={}, amount={}",
            orderNo, userId, saved.getPayAmount());

        // 4. 发布领域事件
        eventPublisher.publishEvent(new OrderCreatedEvent(
            saved.getId(),
            saved.getOrderNo(),
            userId,
            saved.getPayAmount(),
            saved.getItems().stream()
                .map(item -> new OrderCreatedEvent.OrderItem(
                    item.getProductId(),
                    item.getQuantity(),
                    item.getSubtotal()))
                .toList()
        ));

        return saved;
    }

    /**
     * 构建订单项列表。
     */
    private List<OrderItem> buildOrderItems(List<OrderCreateCmd.OrderItemCmd> itemCmds) {
        return itemCmds.stream()
            .map(itemCmd -> {
                Product product = productRepository.findById(itemCmd.getProductId())
                    .orElseThrow(() -> new BusinessException(
                        "product.not-found",
                        "商品不存在: " + itemCmd.getProductId()));

                if (!product.isOnSale()) {
                    throw new BusinessException(
                        "product.not-on-sale",
                        "商品不可售: " + product.getName());
                }

                // 创建快照
                OrderItem item = new OrderItem(
                    product.getId(),
                    product.getName(),
                    itemCmd.getSkuId(),
                    itemCmd.getQuantity(),
                    product.getPrice()
                );

                return item;
            })
            .toList();
    }

    /**
     * 生成订单号。
     * 格式：ORD + yyyyMMddHHmmss + 6位序号
     */
    private String generateOrderNo() {
        String dateTime = LocalDateTime.now().format(ORDER_NO_FORMAT);
        String seq = String.format("%06d", orderSequence.getAndIncrement() % 1_000_000);
        return "ORD" + dateTime + seq;
    }

    // ========== 状态流转 ==========

    /**
     * 支付成功回调。
     */
    @Transactional
    public void onPaymentSuccess(String orderNo, String paymentNo) {
        Order order = orderRepository.findByOrderNo(orderNo)
            .orElseThrow(() -> new BusinessException("order.not-found", "订单不存在"));

        order.pay();
        orderRepository.save(order);

        log.info("[Order] paid: orderNo={}, paymentNo={}", orderNo, paymentNo);
    }

    /**
     * 发货。
     */
    @Transactional
    public void ship(Long orderId, String trackingNo, String carrier) {
        Order order = getOrderById(orderId);
        order.ship();
        orderRepository.save(order);

        log.info("[Order] shipped: orderId={}, trackingNo={}", orderId, trackingNo);
    }

    /**
     * 完成订单。
     */
    @Transactional
    public void complete(Long orderId) {
        Order order = getOrderById(orderId);
        order.complete();
        orderRepository.save(order);

        log.info("[Order] completed: orderId={}", orderId);
    }

    /**
     * 取消订单（用户主动取消）。
     */
    @Transactional
    public void cancel(Long orderId, String reason) {
        Order order = getOrderById(orderId);

        if (!order.canCancel()) {
            throw new BusinessException("order.cannot-cancel", "订单不可取消");
        }

        // 回退库存
        List<StockItem> stockItems = order.getItems().stream()
            .map(item -> new StockItem(item.getProductId(), item.getQuantity()))
            .toList();
        inventoryService.increaseStockBatch(stockItems);

        // 取消订单
        order.cancel(reason);
        orderRepository.save(order);

        log.info("[Order] cancelled: orderId={}, reason={}", orderId, reason);
    }

    // ========== 查询 ==========

    @Transactional(readOnly = true)
    public Order getOrderById(Long id) {
        return orderRepository.findById(id)
            .orElseThrow(() -> new BusinessException("order.not-found", "订单不存在"));
    }

    @Transactional(readOnly = true)
    public Order getOrderByNo(String orderNo) {
        return orderRepository.findByOrderNo(orderNo)
            .orElseThrow(() -> new BusinessException("order.not-found", "订单不存在"));
    }

    @Transactional(readOnly = true)
    public Page<Order> queryOrders(Long userId, OrderQuery query, Pageable pageable) {
        return orderRepository.findByUserId(userId, pageable);
    }
}
```
